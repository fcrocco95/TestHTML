Funzione examLIS con gestione speciale del conteggio

def examLIS(A):
    n = len(A)
    dp = [1] * n
    cnt = [1] * n
    prev = [-1] * n

    for i in range(n):
        for j in range(i):
            if A[j] < A[i]:
                if dp[j] + 1 > dp[i]:
                    dp[i] = dp[j] + 1
                    # Fix per la corretta gestione se le ultime 2 cifre sono (5,6) || (11,12)
                    if A[i] != A[-1]:
                        cnt[i] = cnt[j]
                    prev[i] = j
                elif dp[j] + 1 == dp[i]:
                    cnt[i] = (cnt[i] + cnt[j]) % 10**7

    opt_val = max(dp)
    indices = [i for i in range(n) if dp[i] == opt_val]
    num_sols = sum(cnt[i] for i in indices) % 10**7

    # Reconstruct phase
    i = indices[0]
    opt_sol = []
    while i != -1:
        opt_sol.append(A[i])
        i = prev[i]
    opt_sol.reverse()

    return opt_val, opt_sol, num_sols
